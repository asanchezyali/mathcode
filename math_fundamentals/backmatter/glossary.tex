% glossary.tex - Glossary definitions for Math Fundamentals

\newglossaryentry{conjunto}{
    name={Conjunto},
    description={Colección de objetos bien definidos, llamados elementos, considerados como una entidad única}
}

\newglossaryentry{proposicion}{
    name={Proposición},
    description={Enunciado que puede ser verdadero o falso, pero no ambos}
}

\newglossaryentry{axioma}{
    name={Axioma},
    description={Proposición que se acepta como verdadera sin demostración y sirve como punto de partida para deducir otras proposiciones}
}

\newglossaryentry{teorema}{
    name={Teorema},
    description={Proposición que se puede demostrar a partir de axiomas u otros teoremas previamente demostrados}
}

\newglossaryentry{demostracion}{
    name={Demostración},
    description={Secuencia de pasos lógicos que establecen la verdad de una proposición}
}
